  <script>
    /*

    function changeColor(n){
      for(var i=1; i<=3; i++){
        var list = document.getElementById("list"+i);
        list.style.background = "";
        list.style.color = "";
      }
      var list = document.getElementById("list"+n);
      list.style.background = "#ffffff";
      list.style.color = "#4e4e4e";
    }



    var order = document.querySelector("#totalMoney");
    var pp = 0;
    function addCart(n){
      var tuna = n.id;
      var price = 0;
      if(tuna == "fattyTuna8"){
        price = 22000;
      }if(tuna == "fattyTuna2"){
        price = 10000;
      }if(tuna = "bestPartOfRed3"){
        price = 12000;
      }else {
        price = 16000;
      }
      pp += price;
      order.innerHTML = pp.toLocaleString('ko-KR');
      
    }

    function doOrder() {
      if(pp == 0){
        alert("장바구니에 아무것도 없습니다!");
        return;
      }
      pp = 0;
      var order = document.querySelector("#totalMoney");
      order.innerHTML = "0";
      alert("주문 완료");
    }
    function soldOut(){
      alert("죄송합니다. 참다랑어 세트는 품절입니다.");
      return;
    }
